\section*{GRPO Rollout Benchmark: Metrics and Table}

\subsection*{Speedup Formula}
\[
\text{Speedup}=
\frac{\text{OutputTput}_{\text{quant}}}{\text{OutputTput}_{\text{BF16}}}
\approx
\frac{\text{Elapsed}_{\text{BF16}}}{\text{Elapsed}_{\text{quant}}}
\]
\noindent
Interpretation: $>1.00\times$ indicates improvement, $<1.00\times$ indicates regression.
